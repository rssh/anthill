Proof with explicit query and mapping
=====================================

namespace test
  rule add_comm(?a, ?b) :- eq(add(?a, ?b), add(?b, ?a))
  proof add_comm
    by z3
    query "(assert (forall ((a Int) (b Int)) (= (+ a b) (+ b a))))"
    mapping {
      add -> "+",
      eq -> "=",
      Int -> Int
    }
  end
end
